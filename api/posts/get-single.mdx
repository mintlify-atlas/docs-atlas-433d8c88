---
title: 'Get Single Post'
api: 'GET /api/posts/:id'
description: 'Retrieve a specific post by ID with full details'
---

## Authentication

This endpoint requires JWT authentication. Include the token in the Authorization header:

```
Authorization: Bearer <token>
```

## Path Parameters

<ParamField path="id" type="string" required>
  The MongoDB ObjectId of the post to retrieve
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the post
</ResponseField>

<ResponseField name="title" type="string">
  The title of the post
</ResponseField>

<ResponseField name="content" type="string">
  The content of the post
</ResponseField>

<ResponseField name="media_url" type="string">
  URL path to the media file
</ResponseField>

<ResponseField name="media_type" type="string">
  Type of media: `image` or `video`
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp when the post was created
</ResponseField>

<ResponseField name="analysis_status" type="string">
  Status of deepfake analysis: `none`, `processing`, `completed`, or `failed`
</ResponseField>

<ResponseField name="profiles" type="object">
  Creator information
  
  <ResponseField name="username" type="string">
    Username of the post creator
  </ResponseField>
</ResponseField>

<ResponseField name="deepfake_analysis" type="object" nullable>
  Deepfake analysis results (null if not analyzed)
  
  <ResponseField name="is_fake" type="boolean">
    Whether the video is classified as a deepfake
  </ResponseField>
  
  <ResponseField name="confidence" type="number">
    Average confidence score across all frames (0-1)
  </ResponseField>
  
  <ResponseField name="frames_analysis" type="array">
    Array of frame-level analysis results
    
    <ResponseField name="frame" type="string">
      Frame identifier/number
    </ResponseField>
    
    <ResponseField name="confidence" type="number">
      Confidence score for this frame (0-1)
    </ResponseField>
    
    <ResponseField name="is_fake" type="boolean">
      Whether this frame is classified as fake
    </ResponseField>
    
    <ResponseField name="frame_path" type="string">
      Path to the extracted frame image
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="summary" type="object">
    Summary statistics of the analysis
    
    <ResponseField name="status" type="string">
      Overall classification: `REAL` or `FAKE`
    </ResponseField>
    
    <ResponseField name="confidence_percentage" type="number">
      Confidence percentage (0-100)
    </ResponseField>
    
    <ResponseField name="total_frames" type="number">
      Total number of frames analyzed
    </ResponseField>
    
    <ResponseField name="real_frames" type="number">
      Number of frames classified as real
    </ResponseField>
    
    <ResponseField name="fake_frames" type="number">
      Number of frames classified as fake
    </ResponseField>
  </ResponseField>
</ResponseField>

## Code Examples

<CodeGroup>

```bash cURL
curl http://localhost:3000/api/posts/507f1f77bcf86cd799439011 \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

```javascript JavaScript
const response = await fetch(
  'http://localhost:3000/api/posts/507f1f77bcf86cd799439011',
  {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  }
);

const post = await response.json();
```

```python Python
import requests

url = "http://localhost:3000/api/posts/507f1f77bcf86cd799439011"
headers = {"Authorization": f"Bearer {token}"}

response = requests.get(url, headers=headers)
post = response.json()
```

</CodeGroup>
